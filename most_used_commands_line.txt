ensuite
-- 1️⃣ Créer l'utilisateur avec un mot de passe

-- 2️⃣ Créer la base de données
CREATE DATABASE cofilab_db OWNER cofilab_user;

-- 3️⃣ Donner les droits complets
GRANT ALL PRIVILEGES ON DATABASE cofilab_db TO cofilab_user;



Si tu veux vérifier que tout est bon avant de relancer :
psql -U cofilab_user -d cofilab_db -h localhost
